def openGDALDataset(FileName):
    dataset = gdal.Open(FileName)
    if dataset is None:
        sys.exit("Data set open failed: " + FileName)
    return dataset

def getDatasetProperties(dataset):
    # Get the properties of the rasters in dataset
    xsize = dataset.RasterXSize
    ysize = dataset.RasterYSize
    nbands = dataset.RasterCount
    projection = dataset.GetProjection()
    geotransform = dataset.GetGeoTransform()
    return xsize, ysize, nbands, projection, geotransform

def selectBands(algorithm_WLs, sensor_central_WLs, sensor_width):
    # This function returns the indices of the sensor bands that are closest to 
    # the algorithm's, following the same order as the algorithm wavelengths 
    
    algorithm_bands = [] 
    # for each algorithm wavelength, find the closest sensor band, quit if outside the band's width 
    for algorithm_WL in algorithm_WLs:

        # find the closest sensor central wavelengths 
        diff = [abs(algorithm_WL - central_WL) for central_WL in sensor_central_WLs]
        band_closest = (diff.index(min(diff))) # index of the closest band 

        # if outside the band's width, quit
        if (min(diff) >= sensor_width[band_closest]/2):
            sys.exit("Warning: wavelength {} is not supported".format(algorithm_WL)) 
        algorithm_bands.append(band_closest) 

    # sensor wavelengths used in the algorithm
    sensor_used_WL = [sensor_central_WLs[band] for band in algorithm_bands]
    print("The algorithm uses {}, matching the sensor's {}".format(algorithm_WLs, sensor_used_WL))
    return algorithm_bands

def getAlgorithmInfo(name):
    if name == "Dsa_07":
        algorithm_wavelengths = [671,551] 
        expression = "10**(-0.44 + 1.93 * np.log10(arrays[algorithm_bands[0]]/arrays[algorithm_bands[1]]))"
    elif name == "Doxaran_02": 
        algorithm_wavelengths = [862,551] 
        expression = "np.exp(2.8*arrays[algorithm_bands[0]]/arrays[algorithm_bands[1]] + 3.53)"
    return algorithm_wavelengths, expression

def bandMath(file_in,file_out,algorithm_bands,expression):
    
    # This function takes in a tif, does the band math, mask SWIR>400,
    # and export to tif
    
    ds = openGDALDataset(file_in)
    xsize, ysize, nBands, projection, geotransform = getDatasetProperties(ds)
    bands = []
    for i in range(nBands):
        bands.append(ds.GetRasterBand(i+1))
    arrays = []
    for i in range(nBands):
        arrays.append(bands[i].ReadAsArray(0,0,xsize,ysize))
    
    calculated =  eval(expression)
    
    # mask SWIR>400 as nan
    calculated[arrays[5]>400] = 'nan'

    driver = gdal.GetDriverByName("GTiff")
    file_out = file_out
    dataset = driver.Create(file_out, xsize, ysize, 1, gdal.GDT_Float32)
    dataset.SetGeoTransform(geotransform)
    dataset.SetProjection(projection)
    band = dataset.GetRasterBand(1)
    band.WriteArray(calculated) 
    dataset = None 
    return 0
